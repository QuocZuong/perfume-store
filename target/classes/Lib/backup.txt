CREATE TABLE [Product] (
    Code VARCHAR(10) Not Null,
    [Name] NVARCHAR(50),
    BrandCode VARCHAR(5),
    Price INT,
    Gender NVARCHAR(50),
    Smell NVARCHAR(200),
    Quantity INT DEFAULT 0,
    ReleaseYear SMALLINT DEFAULT 2003,
    Volume INT DEFAULT 100,
    Description NVARCHAR(MAX),

	Primary Key (Code)
);

CREATE TABLE [Brand] (
    Code VARCHAR(5) PRIMARY KEY,
    [Name] NVARCHAR(50)
);

CREATE TABLE [User] (
    ID INT NOT NULL PRIMARY KEY,
    [Name] NVARCHAR(50),
    UserName VARCHAR(50),
    [Password] VARCHAR(32),
    PhoneNumber VARCHAR(10),
    Email VARCHAR(100),
    [Address] NVARCHAR(500)
);

CREATE TABLE [UserProducts] (
    UserID INT NOT NULL,
    ProductCode VARCHAR(10) NOT NULL,
    SellQuantity INT DEFAULT 0
);

CREATE TABLE [Admin] (
    ID INT NOT NULL PRIMARY KEY,
    [Name] NVARCHAR(50),
    [Password] NVARCHAR(50),
    PhoneNumber VARCHAR(10)
);

ALTER TABLE Product
ADD FOREIGN KEY (BrandCode) REFERENCES Brand(Code);

ADD FOREIGN KEY (UserID) REFERENCES [User](ID);
